---
title: Tooltip
description: A popup that displays information when hovering over or focusing an element.
---

import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger, Button } from '@e412/rnui-react';
import { Plus, Settings, Trash2, Copy } from 'lucide-react';

The `Tooltip` component shows contextual information on hover or focus. Wrap your app (or a subtree) in `TooltipProvider` to enable tooltips.

## Import

```tsx
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from '@e412/rnui-react';
```

## Usage

<Preview>
  <TooltipProvider>
    <Tooltip>
      <TooltipTrigger render={<Button variant="outline" />}>
        Hover me
      </TooltipTrigger>
      <TooltipContent>
        <p>Add to library</p>
      </TooltipContent>
    </Tooltip>
  </TooltipProvider>
</Preview>
