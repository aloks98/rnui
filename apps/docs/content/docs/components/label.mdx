import { Label, Input, Slider } from '@e412/rnui-react';

The `Label` component renders an accessible `<label>` element that associates with a form control via the `htmlFor` attribute. It is used alongside inputs, selects, and other form elements.

## Import

```tsx
import { Label } from '@e412/rnui-react';
```

## Usage

<Preview>
  <div className="grid w-full max-w-sm gap-1.5">
    <Label htmlFor="email">Email address</Label>
    <Input type="email" id="email" placeholder="you@example.com" />
  </div>
</Preview>

```tsx
<div className="grid w-full max-w-sm gap-1.5">
  <Label htmlFor="email">Email address</Label>
  <Input type="email" id="email" placeholder="you@example.com" />
</div>
```
