import { ScrollArea, Separator } from '@e412/rnui-react';

The `ScrollArea` component provides a cross-browser consistent scrollable container with custom-styled scrollbars. Use it when you need a scrollable region that looks the same across all browsers and operating systems.

## Import

```tsx
import { ScrollArea } from '@e412/rnui-react';
```

## Usage

export const tags = Array.from({ length: 50 }).map((_, i) => `v1.2.0-beta.${i + 1}`);

<Preview>
  <ScrollArea className="h-72 w-48 rounded-md border">
    <div className="p-4">
      <h4 className="mb-4 text-sm font-medium leading-none">Tags</h4>
      {tags.map((tag) => (
        <div key={tag}>
          <div className="text-sm">{tag}</div>
          <Separator className="my-2" />
        </div>
      ))}
    </div>
  </ScrollArea>
</Preview>

## Examples

### Chat Messages

export const messages = [
  { id: 1, sender: 'Alice', text: 'Hey, how are you?' },
  { id: 2, sender: 'You', text: 'Doing great, thanks!' },
  { id: 3, sender: 'Alice', text: 'Want to grab coffee later?' },
  { id: 4, sender: 'You', text: 'Sure, around 3pm works.' },
  { id: 5, sender: 'Alice', text: 'Perfect, see you then!' },
  { id: 6, sender: 'You', text: 'Looking forward to it.' },
  { id: 7, sender: 'Alice', text: 'By the way, did you see the new update?' },
  { id: 8, sender: 'You', text: 'Not yet, what changed?' },
  { id: 9, sender: 'Alice', text: 'They added dark mode and new themes.' },
  { id: 10, sender: 'You', text: 'Nice, I will check it out.' },
  { id: 11, sender: 'Alice', text: 'Let me know what you think.' },
  { id: 12, sender: 'You', text: 'Will do!' },
];

<Preview>
  <ScrollArea className="h-64 w-80 rounded-md border">
    <div className="p-4 space-y-3">
      {messages.map((msg) => (
        <div key={msg.id} className={`flex flex-col ${msg.sender === 'You' ? 'items-end' : 'items-start'}`}>
          <span className="text-xs text-muted-foreground mb-1">{msg.sender}</span>
          <div className={`rounded-lg px-3 py-2 text-sm max-w-[200px] ${msg.sender === 'You' ? 'bg-primary text-primary-foreground' : 'bg-muted'}`}>
            {msg.text}
          </div>
        </div>
      ))}
    </div>
  </ScrollArea>
</Preview>
