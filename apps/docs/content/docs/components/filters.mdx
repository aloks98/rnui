import { Filters, Button } from '@e412/rnui-react';
import { FilterIcon } from 'lucide-react';

The `Filters` component provides a dynamic filter bar where users can add, modify, and remove filter conditions. It supports text, select, and multiselect field types, preset filter values, and size variants. Filter state is managed externally for full control.

## Import

```tsx
import { Filters, Filter, FilterFieldConfig, createFilter } from '@e412/rnui-react';
```

## Usage

export const fields = [
  {
    key: 'status',
    label: 'Status',
    type: 'select',
    options: [
      { value: 'active', label: 'Active' },
      { value: 'inactive', label: 'Inactive' },
    ],
  },
  {
    key: 'role',
    label: 'Role',
    type: 'multiselect',
    options: [
      { value: 'ceo', label: 'CEO' },
      { value: 'cto', label: 'CTO' },
      { value: 'designer', label: 'Designer' },
      { value: 'developer', label: 'Developer' },
      { value: 'engineer', label: 'Engineer' },
    ],
  },
  {
    key: 'name',
    label: 'Name',
    type: 'text',
  },
  {
    key: 'company',
    label: 'Company',
    type: 'text',
  },
];

export function DefaultFiltersDemo() {
  const [filters, setFilters] = React.useState([]);
  return (
    <Filters
      filters={filters}
      fields={fields}
      onChange={setFilters}
      trigger={
        <Button variant="outline" size="sm">
          <FilterIcon className="size-4" />
          Filter
        </Button>
      }
    />
  );
}

## Field Configuration

Each field in the `fields` array accepts the following shape:

| Property | Type | Description |
| --- | --- | --- |
| `key` | `string` | Unique identifier for the filter field. |
| `label` | `string` | Display label shown in the filter UI. |
| `type` | `'text' \| 'select' \| 'multiselect'` | The type of filter input. |
| `options` | `{ value: string; label: string }[]` | Options for select and multiselect types. |
