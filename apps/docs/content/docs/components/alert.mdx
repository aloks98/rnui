---
title: Alert
description: Callout banners for displaying important messages with semantic color variants for different severity levels.
---

## Installation

```bash
npm install @e412/rnui-react @e412/rnui-themes
```

## Import

```tsx
import { Alert, AlertTitle, AlertDescription, AlertAction } from '@e412/rnui-react'
```

## Usage

```tsx
<Alert>
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can add components to your app using the CLI.
  </AlertDescription>
</Alert>
```

<Preview>
  <Alert className="max-w-lg">
    <InfoIcon className="size-4" />
    <AlertTitle>Heads up!</AlertTitle>
    <AlertDescription>You can add components to your app using the cli.</AlertDescription>
  </Alert>
</Preview>

## Examples

### Default

```tsx
<Alert>
  <AlertTitle>Default Alert</AlertTitle>
  <AlertDescription>This is a default alert with a card background.</AlertDescription>
</Alert>
```

### Destructive

```tsx
import { AlertCircleIcon } from 'lucide-react'

<Alert variant="destructive">
  <AlertCircleIcon />
  <AlertTitle>Error</AlertTitle>
  <AlertDescription>
    Your session has expired. Please log in again.
  </AlertDescription>
</Alert>
```

### Info

```tsx
import { InfoIcon } from 'lucide-react'

<Alert variant="info">
  <InfoIcon />
  <AlertTitle>Information</AlertTitle>
  <AlertDescription>
    A new software update is available for download.
  </AlertDescription>
</Alert>
```

### Success

```tsx
import { CheckCircleIcon } from 'lucide-react'

<Alert variant="success">
  <CheckCircleIcon />
  <AlertTitle>Success</AlertTitle>
  <AlertDescription>
    Your changes have been saved successfully.
  </AlertDescription>
</Alert>
```

### Warning

```tsx
import { TriangleAlertIcon } from 'lucide-react'

<Alert variant="warning">
  <TriangleAlertIcon />
  <AlertTitle>Warning</AlertTitle>
  <AlertDescription>
    Your account is nearing its storage limit.
  </AlertDescription>
</Alert>
```

### Invert

```tsx
<Alert variant="invert">
  <AlertTitle>Notice</AlertTitle>
  <AlertDescription>
    This uses an inverted color scheme for high contrast.
  </AlertDescription>
</Alert>
```

### With Action

```tsx
import { Button } from '@e412/rnui-react'
import { AlertCircleIcon } from 'lucide-react'

<Alert variant="destructive">
  <AlertCircleIcon />
  <AlertTitle>Subscription Expiring</AlertTitle>
  <AlertDescription>
    Your subscription will expire in 3 days.
  </AlertDescription>
  <AlertAction>
    <Button size="sm" variant="outline">Renew</Button>
  </AlertAction>
</Alert>
```

### With Icon Only

```tsx
import { InfoIcon } from 'lucide-react'

<Alert variant="info">
  <InfoIcon />
  <AlertTitle>You have 3 unread messages.</AlertTitle>
</Alert>
```

## Props

### Alert

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "destructive" \| "info" \| "success" \| "warning" \| "invert"` | `"default"` | Visual style variant |
| `className` | `string` | — | Additional CSS classes |

### AlertTitle

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | Title text |
| `className` | `string` | — | Additional CSS classes |

### AlertDescription

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | Description content |
| `className` | `string` | — | Additional CSS classes |

### AlertAction

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | Action elements (e.g. buttons) |
| `className` | `string` | — | Additional CSS classes |
