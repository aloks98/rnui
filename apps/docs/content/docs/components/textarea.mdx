import { Textarea, Label, Button } from '@e412/rnui-react';

The `Textarea` component renders a styled multi-line text input. Use it for comments, messages, descriptions, and other long-form content.

## Import

```tsx
import { Textarea } from '@e412/rnui-react';
```

## Usage

<Preview>
  <Textarea placeholder="Type your message here." className="max-w-sm" />
</Preview>

## With Label

<Preview>
  <div className="grid w-full max-w-sm gap-1.5">
    <Label htmlFor="message">Your message</Label>
    <Textarea placeholder="Type your message here." id="message" />
  </div>
</Preview>

## Disabled

<Preview>
  <Textarea placeholder="Disabled" disabled className="max-w-sm" />
</Preview>
