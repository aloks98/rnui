title: NativeSelect
description: A styled wrapper around the native HTML select element.
---

import { NativeSelect, NativeSelectOption, NativeSelectOptGroup, Label } from '@e412/rnui-react';

The `NativeSelect` component provides a styled native HTML select. It uses the browser's built-in dropdown, making it ideal for mobile devices and simple forms.

## Import

```tsx
import {
  NativeSelect,
  NativeSelectOption,
  NativeSelectOptGroup,
} from '@e412/rnui-react';
```

## Usage

<Preview code={`<div className="grid w-full max-w-sm gap-1.5">
  <Label>Fruit</Label>
  <NativeSelect>
    <NativeSelectOption value="">Select a fruit</NativeSelectOption>
    <NativeSelectOption value="apple">Apple</NativeSelectOption>
    <NativeSelectOption value="banana">Banana</NativeSelectOption>
    <NativeSelectOption value="blueberry">Blueberry</NativeSelectOption>
    <NativeSelectOption value="grapes">Grapes</NativeSelectOption>
    <NativeSelectOption value="pineapple">Pineapple</NativeSelectOption>
  </NativeSelect>
</div>`}>
  <div className="grid w-full max-w-sm gap-1.5">
    <Label>Fruit</Label>
    <NativeSelect>
      <NativeSelectOption value="">Select a fruit</NativeSelectOption>
      <NativeSelectOption value="apple">Apple</NativeSelectOption>
      <NativeSelectOption value="banana">Banana</NativeSelectOption>

      <NativeSelectOption value="grapes">Grapes</NativeSelectOption>
      <NativeSelectOption value="pineapple">Pineapple</NativeSelectOption>

</Preview>
