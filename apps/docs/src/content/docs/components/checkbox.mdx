---
title: Checkbox
description: A checkbox input component built on Base UI's Checkbox primitive.
---

## Installation

```bash
npx @e412/rnui-cli add checkbox
```

## Import

```tsx
import { Checkbox } from '@e412/rnui-react'
```

## Usage

```tsx
<Checkbox />
```

## Examples

### With Label

```tsx
import { Label } from '@e412/rnui-react'

<div className="flex items-center gap-2">
  <Checkbox id="terms" />
  <Label htmlFor="terms">Accept terms and conditions</Label>
</div>
```

### Checked by Default

```tsx
<Checkbox defaultChecked />
```

### Controlled

```tsx
const [checked, setChecked] = React.useState(false)

<Checkbox checked={checked} onCheckedChange={setChecked} />
```

### Disabled

```tsx
<Checkbox disabled />
<Checkbox disabled defaultChecked />
```

### With Form Field

```tsx
import { Field, FieldLabel, FieldDescription } from '@e412/rnui-react'

<Field>
  <div className="flex items-center gap-2">
    <Checkbox id="notifications" />
    <FieldLabel htmlFor="notifications">Enable notifications</FieldLabel>
  </div>
  <FieldDescription>
    Receive email notifications about updates.
  </FieldDescription>
</Field>
```

### Invalid State

```tsx
<Checkbox aria-invalid="true" />
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `checked` | `boolean` | — | Controlled checked state |
| `defaultChecked` | `boolean` | `false` | Initial checked state (uncontrolled) |
| `onCheckedChange` | `(checked: boolean) => void` | — | Callback when checked state changes |
| `disabled` | `boolean` | `false` | Disabled state |
| `required` | `boolean` | `false` | Required for form validation |
| `name` | `string` | — | Form field name |
| `value` | `string` | `"on"` | Form field value |
| `className` | `string` | — | Additional CSS classes |
