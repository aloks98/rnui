---
title: Textarea
description: A multi-line text input component with automatic field sizing.
---

## Installation

```bash
npm install @e412/rnui-react @e412/rnui-themes
```

## Import

```tsx
import { Textarea } from '@e412/rnui-react'
```

## Usage

```tsx
<Textarea placeholder="Type your message here." />
```

<Preview>
  <div className="grid w-full max-w-sm gap-1.5">
    <Label htmlFor="message">Message</Label>
    <Textarea id="message" placeholder="Type your message..." />
  </div>
</Preview>

## Examples

### With Label

```tsx
<div className="grid gap-1.5">
  <label htmlFor="message">Your message</label>
  <Textarea id="message" placeholder="Type your message here." />
</div>
```

### Disabled

```tsx
<Textarea placeholder="Disabled textarea" disabled />
```

### With Default Value

```tsx
<Textarea defaultValue="This is some pre-filled text content." />
```

### Custom Rows

```tsx
<Textarea placeholder="Fixed height textarea" rows={6} />
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `placeholder` | `string` | - | Placeholder text |
| `defaultValue` | `string` | - | Default text content |
| `value` | `string` | - | Controlled text content |
| `onChange` | `(e: ChangeEvent) => void` | - | Callback when text changes |
| `rows` | `number` | - | Number of visible text rows |
| `disabled` | `boolean` | `false` | Disable the textarea |
| `className` | `string` | - | Additional CSS classes |
