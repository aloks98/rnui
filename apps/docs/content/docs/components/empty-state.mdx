---
title: EmptyState
description: A pre-composed empty state with icon, title, description, and optional action.
---

import { EmptyState, Button } from '@e412/rnui-react';
import { Inbox, Plus, Search, FileText } from 'lucide-react';

The `EmptyState` component provides a ready-made empty state pattern with an icon, title, description, and optional action button. Use it when a list, table, or section has no data to show.

## Import

```tsx
import { EmptyState } from '@e412/rnui-react';
```

## Usage

<Preview>
  <EmptyState
    icon={<Inbox />}
    title="No messages"
    description="You have no messages in your inbox. New messages will appear here."
  />
</Preview>

```tsx
<EmptyState
    icon={<Inbox />}
    title="No messages"
    description="You have no messages in your inbox. New messages will appear here."
  />
```

## With Action

<Preview>
  <EmptyState
    icon={<Inbox />}
    title="No projects yet"
    description="Get started by creating your first project."
    action={
      <Button size="sm">
        <Plus className="mr-2 h-4 w-4" />
        Create Project
      </Button>
    }
  />
</Preview>

```tsx
<EmptyState
    icon={<Inbox />}
    title="No projects yet"
    description="Get started by creating your first project."
    action={
      <Button size="sm">
        <Plus className="mr-2 h-4 w-4" />
        Create Project
      </Button>
    }
  />
```
