---
title: Input
description: A styled text input component built on Base UI Input.
---

## Installation

```bash
npx @e412/rnui-cli add input
```

## Import

```tsx
import { Input } from '@e412/rnui-react'
```

## Usage

```tsx
<Input placeholder="Enter text..." />
```

## Examples

### Default

```tsx
<Input placeholder="Email address" />
```

### With Type

```tsx
<Input type="email" placeholder="you@example.com" />
<Input type="password" placeholder="Password" />
<Input type="number" placeholder="0" />
```

### File Input

```tsx
<Input type="file" />
```

### Disabled

```tsx
<Input placeholder="Disabled" disabled />
```

### Invalid State

```tsx
<Input placeholder="Invalid input" aria-invalid="true" />
```

### With Label

```tsx
import { Label } from '@e412/rnui-react'

<div className="grid gap-2">
  <Label htmlFor="name">Name</Label>
  <Input id="name" placeholder="Your name" />
</div>
```

### With Form

```tsx
<FormField
  control={form.control}
  name="email"
  render={({ field }) => (
    <FormItem>
      <FormLabel>Email</FormLabel>
      <FormControl>
        <Input placeholder="you@example.com" {...field} />
      </FormControl>
      <FormMessage />
    </FormItem>
  )}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `string` | - | HTML input type (`text`, `email`, `password`, etc.) |
| `placeholder` | `string` | - | Placeholder text |
| `disabled` | `boolean` | `false` | Disabled state |
| `className` | `string` | - | Additional CSS class |

The `Input` component forwards all standard HTML `<input>` props.
