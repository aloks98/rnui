import { Table, TableHeader, TableBody, TableRow, TableHead, TableCell, TableCaption, Badge } from '@e412/rnui-react';

The `Table` component renders a semantic HTML `<table>` with consistent styling. Use it for static, read-only tabular data. For interactive tables with sorting, filtering, and pagination, see [DataGrid](/docs/components/data-grid).

## Import

```tsx
import {
  Table,
  TableHeader,
  TableBody,
  TableRow,
  TableHead,
  TableCell,
  TableCaption,
} from '@e412/rnui-react';
```
