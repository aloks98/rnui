## Installation

```bash
npm install @e412/rnui-react @e412/rnui-themes
```

## Import

```tsx
import { Calendar } from '@e412/rnui-react'
```

## Usage

```tsx
const [date, setDate] = React.useState<Date | undefined>(new Date())

<Calendar
  mode="single"
  selected={date}
  onSelect={setDate}
/>
```

<Preview>
  <Calendar className="rounded-md border" />
</Preview>

## Examples

### Single Date Selection

```tsx
const [date, setDate] = React.useState<Date | undefined>(new Date())

<Calendar
  mode="single"
  selected={date}
  onSelect={setDate}
/>
```

### Date Range Selection

```tsx
const [range, setRange] = React.useState<DateRange | undefined>({
  from: new Date(),
  to: addDays(new Date(), 7),
})

<Calendar
  mode="range"
  selected={range}
  onSelect={setRange}
  numberOfMonths={2}
/>
```

### With Dropdown Navigation

```tsx
<Calendar
  mode="single"
  captionLayout="dropdown"
  fromYear={2000}
  toYear={2030}
/>
```

### Hide Outside Days

```tsx
<Calendar
  mode="single"
  showOutsideDays={false}
/>
```

### With Week Numbers

```tsx
<Calendar
  mode="single"
  showWeekNumber
/>
```

### Custom Button Variant

```tsx
<Calendar
  mode="single"
  buttonVariant="outline"
/>
```

### Disabled Dates

```tsx
<Calendar
  mode="single"
  disabled={{ before: new Date() }}
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `mode` | `"single" \| "range" \| "multiple"` | — | Selection mode |
| `selected` | `Date \| DateRange \| Date[]` | — | Currently selected date(s) |
| `onSelect` | `(date: ...) => void` | — | Callback when selection changes |
| `showOutsideDays` | `boolean` | `true` | Show days from adjacent months |
| `captionLayout` | `"label" \| "dropdown"` | `"label"` | Month/year caption layout |
| `buttonVariant` | `"default" \| "secondary" \| "destructive" \| "outline" \| "ghost" \| "link"` | `"ghost"` | Navigation button variant |
| `numberOfMonths` | `number` | `1` | Number of months to display |
| `showWeekNumber` | `boolean` | `false` | Show week numbers |
| `disabled` | `Matcher \| Matcher[]` | — | Dates to disable |
| `fromYear` | `number` | — | Start year for dropdown navigation |
| `toYear` | `number` | — | End year for dropdown navigation |
| `locale` | `Locale` | — | Locale for date formatting |
| `className` | `string` | — | Additional CSS classes |
