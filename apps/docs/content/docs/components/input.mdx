import { Input, Label, Button } from '@e412/rnui-react';

The `Input` component renders a styled text input. It supports all standard HTML input types and can be combined with `Label` for accessible forms.

## Import

```tsx
import { Input } from '@e412/rnui-react';
```

## Usage

<Preview code={`<Input type="email" placeholder="Email" className="max-w-sm" />`}>
  <Input type="email" placeholder="Email" className="max-w-sm" />
</Preview>

## With Label

<Preview code={`<div className="grid w-full max-w-sm gap-1.5">
  <Label htmlFor="email">Email</Label>
  <Input type="email" id="email" placeholder="you@example.com" />
</div>`}>
  <div className="grid w-full max-w-sm gap-1.5">
    <Label htmlFor="email">Email</Label>
    <Input type="email" id="email" placeholder="you@example.com" />
  </div>
</Preview>
