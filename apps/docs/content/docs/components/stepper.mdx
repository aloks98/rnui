---
title: Stepper
description: A multi-step indicator showing progress through a sequential workflow.
---

import { Stepper, StepperItem, StepperTrigger, StepperIndicator, StepperSeparator, StepperTitle, StepperDescription, StepperNav, StepperPanel, StepperContent } from '@e412/rnui-react';
import { CheckIcon, LoaderCircleIcon } from 'lucide-react';

The `Stepper` component guides users through a multi-step process. It supports horizontal and vertical orientations, custom indicators (completed, loading), titles, descriptions, and both controlled and uncontrolled modes.

## Import

```tsx
import {
  Stepper,
  StepperItem,
  StepperTrigger,
  StepperIndicator,
  StepperSeparator,
  StepperTitle,
  StepperDescription,
  StepperNav,
  StepperPanel,
  StepperContent,
} from '@e412/rnui-react';
```

## Usage

<Preview>
  <Stepper defaultValue={2} className="w-full max-w-md space-y-8">
    <StepperNav>
      {[1, 2, 3, 4].map((step) => (
        <StepperItem key={step} step={step}>
          <StepperTrigger>
            <StepperIndicator>{step}</StepperIndicator>
          </StepperTrigger>
          {step < 4 && (
            <StepperSeparator className="group-data-[state=completed]/step:bg-primary" />
          )}
        </StepperItem>
      ))}
    </StepperNav>
    <StepperPanel className="text-sm">
      {[1, 2, 3, 4].map((step) => (
        <StepperContent key={step} value={step} className="flex items-center justify-center">
          Step {step} content
        </StepperContent>
      ))}
    </StepperPanel>
  </Stepper>
</Preview>

## Vertical Orientation

<Preview>
  <div className="flex items-center justify-center">
    <Stepper
      className="flex flex-col items-center justify-center gap-10"
      defaultValue={2}
      orientation="vertical"
      indicators={{
        completed: <CheckIcon className="size-3.5" />,
        loading: <LoaderCircleIcon className="size-3.5 animate-spin" />,
      }}
    >
      <StepperNav>
        {[1, 2, 3].map((step) => (
          <StepperItem key={step} step={step} loading={step === 2}>
            <StepperTrigger>
              <StepperIndicator className="data-[state=completed]:bg-green-500 data-[state=completed]:text-white">
                {step}
              </StepperIndicator>
            </StepperTrigger>
            {step < 3 && (
              <StepperSeparator className="group-data-[state=completed]/step:bg-green-500" />
            )}
          </StepperItem>
        ))}
      </StepperNav>
      <StepperPanel className="w-56 text-center text-sm">
        {[1, 2, 3].map((step) => (
          <StepperContent key={step} value={step}>
            Step {step} content
          </StepperContent>
        ))}
      </StepperPanel>
    </Stepper>
  </div>
</Preview>
