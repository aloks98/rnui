import { Empty, EmptyHeader, EmptyMedia, EmptyTitle, EmptyDescription, EmptyContent, Button } from '@e412/rnui-react';
import { Inbox, FolderOpen, Plus, Search } from 'lucide-react';

The `Empty` component provides a fully composable empty state layout. Unlike `EmptyState` (which is a single pre-composed component), `Empty` gives you fine-grained control over each section -- media, header, title, description, and content.

## Import

```tsx
import {
  Empty,
  EmptyHeader,
  EmptyMedia,
  EmptyTitle,
  EmptyDescription,
  EmptyContent,
} from '@e412/rnui-react';
```

## Usage

<Preview code={`<Empty>
    <EmptyHeader>
      <EmptyMedia>
        <Inbox className="h-10 w-10 text-muted-foreground" />
      </EmptyMedia>
      <EmptyTitle>No items found</EmptyTitle>
      <EmptyDescription>There are no items to display right now.</EmptyDescription>
    </EmptyHeader>
  </Empty>`}>
  <Empty>
    <EmptyHeader>
      <EmptyMedia>
        <Inbox className="h-10 w-10 text-muted-foreground" />
      </EmptyMedia>
      <EmptyTitle>No items found</EmptyTitle>
      <EmptyDescription>There are no items to display right now.</EmptyDescription>
    </EmptyHeader>
  </Empty>
</Preview>

## Icon Variant

Use `variant="icon"` on `EmptyMedia` for a contained icon style with a muted background.

<Preview code={`<Empty>
    <EmptyHeader>
      <EmptyMedia variant="icon">
        <FolderOpen />
      </EmptyMedia>
      <EmptyTitle>No files</EmptyTitle>
      <EmptyDescription>Upload files to see them here.</EmptyDescription>
    </EmptyHeader>
  </Empty>`}>
  <Empty>
    <EmptyHeader>
      <EmptyMedia variant="icon">
        <FolderOpen />
      </EmptyMedia>
      <EmptyTitle>No files</EmptyTitle>
      <EmptyDescription>Upload files to see them here.</EmptyDescription>
    </EmptyHeader>
  </Empty>
</Preview>

## With Action Content

Use `EmptyContent` to add action buttons or other interactive elements.

<Preview code={`<Empty>
    <EmptyHeader>
      <EmptyMedia variant="icon">
        <Inbox />
      </EmptyMedia>
      <EmptyTitle>No projects yet</EmptyTitle>
      <EmptyDescription>Create your first project to get started.</EmptyDescription>
    </EmptyHeader>
    <EmptyContent>
      <Button size="sm">
        <Plus className="mr-2 h-4 w-4" />
        Create Project
      </Button>
    </EmptyContent>
  </Empty>`}>
  <Empty>
    <EmptyHeader>
      <EmptyMedia variant="icon">
        <Inbox />
      </EmptyMedia>
      <EmptyTitle>No projects yet</EmptyTitle>
      <EmptyDescription>Create your first project to get started.</EmptyDescription>
    </EmptyHeader>
    <EmptyContent>
      <Button size="sm">
        <Plus className="mr-2 h-4 w-4" />
        Create Project
      </Button>
    </EmptyContent>
  </Empty>
</Preview>
