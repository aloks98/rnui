## Installation

```bash
npm install @e412/rnui-react @e412/rnui-themes
```

## Import

```tsx
import {
  DataGrid,
  DataGridContainer,
  useDataGrid,
} from '@e412/rnui-react'
```

Additional sub-components are available for specific features:

```tsx
import { DataGridTable } from '@e412/rnui-react'
import { DataGridTableVirtual } from '@e412/rnui-react'
import { DataGridTableDnd } from '@e412/rnui-react'
import { DataGridTableDndRows } from '@e412/rnui-react'
import { DataGridPagination } from '@e412/rnui-react'
import { DataGridColumnVisibility } from '@e412/rnui-react'
import { DataGridColumnHeader } from '@e412/rnui-react'
import { DataGridColumnFilter } from '@e412/rnui-react'
import { DataGridScrollArea } from '@e412/rnui-react'
```

## Usage

```tsx
import {
  useReactTable,
  getCoreRowModel,
  createColumnHelper,
} from '@tanstack/react-table'

const columnHelper = createColumnHelper<User>()

const columns = [
  columnHelper.accessor('name', { header: 'Name' }),
  columnHelper.accessor('email', { header: 'Email' }),
  columnHelper.accessor('role', { header: 'Role' }),
]

function UsersTable() {
  const table = useReactTable({
    data,
    columns,
    getCoreRowModel: getCoreRowModel(),
  })

  return (
    <DataGrid table={table} recordCount={data.length}>
      <DataGridContainer>
        <DataGridTable />
      </DataGridContainer>
    </DataGrid>
  )
}
```

### Virtual Scrolling

For large datasets, use `DataGridTableVirtual` for virtualized rendering.

```tsx
<DataGrid table={table} recordCount={data.length}>
  <DataGridContainer>
    <DataGridScrollArea style={{ height: 400 }}>
      <DataGridTableVirtual />
    </DataGridScrollArea>
  </DataGridContainer>
</DataGrid>
```

### Dense Layout

```tsx
<DataGrid
  table={table}
  recordCount={data.length}
  tableLayout={{ dense: true, stripped: true, headerBackground: true }}
>
  <DataGridContainer>
    <DataGridTable />
  </DataGridContainer>
</DataGrid>
```

### Loading State

```tsx
<DataGrid
  table={table}
  recordCount={0}
  isLoading
  loadingMode="skeleton"
  loadingMessage="Loading data..."
  emptyMessage="No records found."
>
  <DataGridContainer>
    <DataGridTable />
  </DataGridContainer>
</DataGrid>
```

## Props

### DataGrid

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `table` | `Table<TData>` | ‚Äî | TanStack Table instance (required) |
| `recordCount` | `number` | ‚Äî | Total number of records |
| `isLoading` | `boolean` | `false` | Show loading state |
| `loadingMode` | `"skeleton" \| "spinner"` | `"skeleton"` | Loading indicator style |
| `loadingMessage` | `ReactNode` | ‚Äî | Custom loading message |
| `emptyMessage` | `ReactNode` | ‚Äî | Message when no data is available |
| `fetchingMoreMessage` | `ReactNode` | ‚Äî | Message while fetching more rows |
| `allRowsLoadedMessage` | `ReactNode` | ‚Äî | Message when all rows are loaded |
| `onRowClick` | `(row: TData) => void` | ‚Äî | Callback when a row is clicked |
| `tableLayout` | `object` | ‚Äî | Layout configuration (see below) |
| `tableClassNames` | `object` | ‚Äî | Custom class names for table parts |
| `className` | `string` | ‚Äî | Additional CSS classes |

### tableLayout Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `dense` | `boolean` | `false` | Compact row height |
| `cellBorder` | `boolean` | `false` | Show cell borders |
| `rowBorder` | `boolean` | `true` | Show row borders |
| `rowRounded` | `boolean` | `false` | Rounded row corners |
| `stripped` | `boolean` | `false` | Alternating row backgrounds |
| `headerBackground` | `boolean` | `true` | Background color on header |
| `headerBorder` | `boolean` | `true` | Border below header |
| `headerSticky` | `boolean` | `false` | Sticky header on scroll |
| `width` | `"auto" \| "fixed"` | `"fixed"` | Table width mode |
| `columnsVisibility` | `boolean` | `false` | Enable column visibility toggle |
| `columnsResizable` | `boolean` | `false` | Enable column resizing |
| `columnsResizeMode` | `"onChange" \| "onEnd"` | `"onEnd"` | When to apply resize |
| `columnsPinnable` | `boolean` | `false` | Enable column pinning |
| `columnsMovable` | `boolean` | `false` | Enable column reordering |
| `columnsDraggable` | `boolean` | `false` | Enable drag-and-drop columns |
| `rowsDraggable` | `boolean` | `false` | Enable drag-and-drop rows |
| `rowsPinnable` | `boolean` | `false` | Enable row pinning |

### DataGridContainer

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `border` | `boolean` | `true` | Show border around the grid |
| `className` | `string` | ‚Äî | Additional CSS classes |
